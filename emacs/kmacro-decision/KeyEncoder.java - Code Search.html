package com.google.storage.graph.bfg.util;

import static java.nio.charset.StandardCharsets.UTF_8;

import com.google.common.hash.LegacyHashing;

import java.nio.ByteBuffer;

/**
 * An encoder for keys in BFG's SSTables and BigTables.
 *
 * @author rictic@google.com (Peter Burns)
 */
public final class KeyEncoder {

  private KeyEncoder() { }

  /**
   * Take all the parts of a triple, hash together, and concatenate the hashes
   * The result will be a byte array of packed longs, in the form:
   *
   * <value1 hash><value2 hash>...<triplehash>
   *
   * @param tripleHash The hash of the triple.
   * @param values String values to hash.
   * @return A byte array of hashed String values.
   */
  public static byte[] makeTripleKey(long tripleHash, String... values) {
    int len = values.length + 1;
    long[] result = new long[len];

    for (int i = 0; i < values.length; i++) {
      result[i] = hashString(values[i]);
    }
    result[len - 1] = tripleHash;

    return packHashes(result);
  }

  /**
   * Create a raw byte array for use in an index. The actual values
   * passed in depend on the index. For example, For a (subject,
   * predicate) prefix in an SPO index, this would be:
   *
   * {@code Triples.makeTripleKeyPrefix(subject, predicate);}
   *
   * @param values String values to hash.
   * @return A byte array of hashed String values.
   */
  public static byte[] makeTripleKeyPrefix(String... values) {
    long[] result = new long[values.length];

    for (int i = 0; i < values.length; i++) {
      result[i] = hashString(values[i]);
    }
    return packHashes(result);
  }

  /**
   * Pack an array of hashes (longs) into a byte array.
   *
   * @param hashes The longs to pack.
   * @return A byte array of the packed longs.
   */
  public static byte[] packHashes(long... hashes) {
    // assuming sizeof(long) == 8 as an optimization
    byte[] bufferArray = new byte[hashes.length * 8];
    ByteBuffer buffer = ByteBuffer.wrap(bufferArray);
    for (long hash : hashes) {
      buffer.putLong(hash);
    }
    return bufferArray;
  }

  /**
   * Generate a component hash of a subject, object, or predicate.
   *
   * <p>This is a convenience method if you wish to hash the
   * components yourself and keep them as longs.
   *
   * @param value subject, object, or predicate as a UTF-8 string.
   * @return A hash of the component.
   */
  public static long hashString(String value) {
    return LegacyHashing.fingerprint1999().hashString(value, UTF_8).asLong();
  }

  /**
   * Partition a BFG keyspace into an arbitrary number of shards,
   * using the default format, which is a leading, single byte of 0 to
   * indicate the triple is 'alive'
   *
   * @see getEvenShardSplits(int, byte[])
   */
  public static byte[][] getEvenShardSplits(int splitCount) {
    return getEvenShardSplits(splitCount, new byte[]{0});
  }

  /**
   * Partition a BFG keyspace into an arbitrary number of shards,
   * returning the a list of the shard boundaries. This will also
   * include a leading byte that is used for "liveness" to do higher
   * levels of grouping.

   * <p>For example, splitting a keyspace into 4 shards with no prefix
   * would result in boundaries at ['\x40', '\x80', '\xc0']
   *
   * @param splitCount the number of partitions
   * @param prefix an optional prefix before the split boundaries
   * @return an array of byte buffers of length splitCount - 1
   */
  public static byte[][] getEvenShardSplits(int splitCount, byte[] prefix) {
    double splitBits = Math.log(splitCount) / Math.log(2);

    int splitBytes = new Double(Math.ceil(splitBits / 8)).intValue();

    // account for both the prefix and the actual boundary value
    int rangeSize = splitBytes + prefix.length;
    byte[][] splitRanges = new byte[splitCount - 1][rangeSize];

    double increment = Math.pow(2, splitBytes * 8) / splitCount;

    for (int i = 0; i < splitCount - 1; i++) {
      double currentSplit = (i + 1) * increment;

      int nextSplit = (int) currentSplit;

      // copy over the prefix
      for (int j = 0; j < prefix.length; j++) {
        splitRanges[i][j] = prefix[j];  // leading 'live' byte
      }

      for (int j = (rangeSize - 1); j >= prefix.length; j--) {
        splitRanges[i][j] = (byte) (nextSplit % 256);
        nextSplit = nextSplit / 256;
      }
    }

    return splitRanges;
  }
}
